# Integration Guide

Learn how to integrate 1sh URL shortener into your applications with best practices and code examples.

## Getting Started

### 1. Understanding the Token System

1sh uses a simple token-based system that doesn't require user accounts:

- **No API keys needed** for most operations
- **Management tokens** are returned when creating links
- **Store tokens securely** to manage your links later

### 2. Implementing Turnstile Verification

For link creation, you'll need to implement Cloudflare Turnstile:

```html
<!-- Add to your HTML head -->
```

## JavaScript/TypeScript Integration

### Basic Link Creation

```typescript
interface CreateLinkRequest {
  url: string;
  customAlias?: string;
  expiresAt?: string;
  turnstileToken: string;
}

interface CreateLinkResponse {
  success: boolean;
  data: {
    id: string;
    shortCode: string;
    shortUrl: string;
    originalUrl: string;
    managementToken: string;
    expiresAt?: string;
    createdAt: string;
  };
}

async function createShortLink(request: CreateLinkRequest): Promise<CreateLinkResponse> {
  const response = await fetch('https://1sh.dev/api/v1/link', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(request),
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error || 'Failed to create link');
  }

  return response.json();
}

// Usage
async function example() {
  try {
    const turnstileToken = await getTurnstileToken();
    
    const link = await createShortLink({
      url: 'https://example.com/very/long/url',
      customAlias: 'my-link',
      turnstileToken,
    });

    console.log('Short URL:', link.data.shortUrl);
    console.log('Management Token:', link.data.managementToken);
    
    // Store the management token for later use
    localStorage.setItem(`link_${link.data.id}`, link.data.managementToken);
  } catch (error) {
    console.error('Error creating link:', error);
  }
}
```

### Link Management Class

```typescript
class LinkManager {
  private baseUrl = 'https://1sh.dev/api/v1';
  
  async createLink(url: string, options: {
    customAlias?: string;
    expiresAt?: Date;
    turnstileToken: string;
  }) {
    const body: any = {
      url,
      turnstileToken: options.turnstileToken,
    };
    
    if (options.customAlias) {
      body.customAlias = options.customAlias;
    }
    
    if (options.expiresAt) {
      body.expiresAt = options.expiresAt.toISOString();
    }
    
    const response = await fetch(`${this.baseUrl}/link`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    
    return this.handleResponse(response);
  }
  
  async getLinks(tokens: string[]) {
    const response = await fetch(
      `${this.baseUrl}/links?tokens=${tokens.join(',')}`
    );
    return this.handleResponse(response);
  }
  
  async getAnalytics(linkId: string, token: string, period = '7d') {
    const response = await fetch(
      `${this.baseUrl}/analytics/${linkId}?token=${token}&period=${period}`
    );
    return this.handleResponse(response);
  }
  
  async checkAlias(alias: string) {
    const response = await fetch(
      `${this.baseUrl}/alias/availability?alias=${encodeURIComponent(alias)}`
    );
    return this.handleResponse(response);
  }
  
  private async handleResponse(response: Response) {
    const data = await response.json();
    
    if (!response.ok) {
      throw new Error(data.error || `HTTP ${response.status}`);
    }
    
    return data;
  }
}

// Usage
const linkManager = new LinkManager();

// Create and manage links
const link = await linkManager.createLink('https://example.com', {
  customAlias: 'example',
  turnstileToken: 'token...',
});

const analytics = await linkManager.getAnalytics(
  link.data.id, 
  link.data.managementToken
);
```

## React Integration

### Custom Hook for Link Management

```tsx
import { useState, useCallback } from 'react';

interface Link {
  id: string;
  shortCode: string;
  shortUrl: string;
  originalUrl: string;
  managementToken: string;
  clickCount?: number;
  createdAt: string;
}

export function useLinks() {
  const [links, setLinks] = useState<Link[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const createLink = useCallback(async (
    url: string,
    options: { customAlias?: string; expiresAt?: Date; turnstileToken: string }
  ) => {
    setLoading(true);
    setError(null);

    try {
      const response = await fetch('/api/v1/link', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          url,
          customAlias: options.customAlias,
          expiresAt: options.expiresAt?.toISOString(),
          turnstileToken: options.turnstileToken,
        }),
      });

      const data = await response.json();

      if (!data.success) {
        throw new Error(data.error);
      }

      const newLink = data.data;
      setLinks(prev => [newLink, ...prev]);
      
      // Store management token
      localStorage.setItem(`link_${newLink.id}`, newLink.managementToken);
      
      return newLink;
    } catch (err) {
      const message = err instanceof Error ? err.message : 'Failed to create link';
      setError(message);
      throw err;
    } finally {
      setLoading(false);
    }
  }, []);

  const loadLinks = useCallback(async () => {
    const tokens = Object.keys(localStorage)
      .filter(key => key.startsWith('link_'))
      .map(key => localStorage.getItem(key))
      .filter(Boolean);

    if (tokens.length === 0) return;

    setLoading(true);
    try {
      const response = await fetch(`/api/v1/links?tokens=${tokens.join(',')}`);
      const data = await response.json();
      
      if (data.success) {
        setLinks(data.data);
      }
    } catch (err) {
      setError('Failed to load links');
    } finally {
      setLoading(false);
    }
  }, []);

  return {
    links,
    loading,
    error,
    createLink,
    loadLinks,
  };
}
```

### Link Creation Component

```tsx
import { useState } from 'react';
import { useLinks } from './useLinks';

export function LinkCreator() {
  const [url, setUrl] = useState('');
  const [customAlias, setCustomAlias] = useState('');

    try {
      const link = await createLink(url, {
        customAlias: customAlias || undefined,
        turnstileToken,
      });

      setUrl('');
      setCustomAlias('');

      // Error is handled by the hook
    }

          type="url"
          value={url}
          onChange={(e) => setUrl(e.target.value)}
          required
          className="w-full p-2 border rounded"
        />

      <div>
        <label htmlFor="alias">Custom alias (optional):</label>
```

## Python Integration

### Basic Python Client

```python
import requests
from typing import Optional, Dict, Any
from datetime import datetime

class OneshClient:
    def __init__(self, base_url: str = "https://1sh.dev/api/v1"):
        self.base_url = base_url
        self.session = requests.Session()
    
    def create_link(
        self,
        url: str,
        custom_alias: Optional[str] = None,
        expires_at: Optional[datetime] = None,
        turnstile_token: str = None
    ) -> Dict[str, Any]:
        """Create a new short link."""
        data = {
            "url": url,
            "turnstileToken": turnstile_token
        }
        
        if custom_alias:
            data["customAlias"] = custom_alias
        
        if expires_at:
            data["expiresAt"] = expires_at.isoformat()
        
        response = self.session.post(f"{self.base_url}/link", json=data)
        response.raise_for_status()
        
        return response.json()
    
    def get_links(self, tokens: list[str]) -> Dict[str, Any]:
        """Get multiple links by management tokens."""
        params = {"tokens": ",".join(tokens)}
        response = self.session.get(f"{self.base_url}/links", params=params)
        response.raise_for_status()
        
        return response.json()
    
    def get_analytics(
        self,
        link_id: str,
        token: str,
        period: str = "7d"
    ) -> Dict[str, Any]:
        """Get analytics for a link."""
        params = {"token": token, "period": period}
        response = self.session.get(
            f"{self.base_url}/analytics/{link_id}",
            params=params
        )
        response.raise_for_status()
        
        return response.json()
    
    def check_alias(self, alias: str) -> Dict[str, Any]:
        """Check if an alias is available."""
        params = {"alias": alias}
        response = self.session.get(
            f"{self.base_url}/alias/availability",
            params=params
        )
        response.raise_for_status()
        
        return response.json()

# Usage example
client = OneshClient()

# Create a link
link = client.create_link(
    url="https://example.com/long-url",
    custom_alias="my-link",
    turnstile_token="demo-token"
)

print(f"Short URL: {link['data']['shortUrl']}")
print(f"Management Token: {link['data']['managementToken']}")

# Get analytics
analytics = client.get_analytics(
    link_id=link['data']['id'],
    token=link['data']['managementToken']
)

print(f"Total clicks: {analytics['data']['analytics']['totalClicks']}")
```

## Error Handling Best Practices

### Comprehensive Error Handling

```typescript
class ApiError extends Error {
  constructor(
    message: string,
    public code: string,
    public status: number
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

async function apiRequest(url: string, options: RequestInit = {}) {
  try {
    const response = await fetch(url, {
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      },
      ...options,
    });

    const data = await response.json();

    if (!response.ok) {
      throw new ApiError(
        data.error || 'Request failed',
        data.code || 'UNKNOWN_ERROR',
        response.status
      );
    }

    return data;
  } catch (error) {
    if (error instanceof ApiError) {
      throw error;
    }
    
    // Network or parsing error
    throw new ApiError(
      'Network error occurred',
      'NETWORK_ERROR',
      0
    );
  }
}

// Usage with specific error handling
try {
  const link = await apiRequest('/api/v1/link', {
    method: 'POST',
    body: JSON.stringify({ url, turnstileToken }),
  });
} catch (error) {
  if (error instanceof ApiError) {
    switch (error.code) {
      case 'ALIAS_TAKEN':
        alert('That alias is already taken. Please choose another.');
        break;
      case 'INVALID_URL':
        alert('Please enter a valid URL.');
        break;
      case 'RATE_LIMIT_EXCEEDED':
        alert('Too many requests. Please wait a moment.');
        break;
      default:
        alert(`Error: ${error.message}`);
    }
  } else {
    alert('An unexpected error occurred.');
  }
}
```

## Security Considerations

### Token Storage

```typescript
// Secure token storage
class TokenManager {
  private static readonly PREFIX = 'onesh_token_';
  
  static store(linkId: string, token: string) {
    // Use sessionStorage for temporary storage
    // or localStorage for persistent storage
    localStorage.setItem(`${this.PREFIX}${linkId}`, token);
  }
  
  static get(linkId: string): string | null {
    return localStorage.getItem(`${this.PREFIX}${linkId}`);
  }
  
  static remove(linkId: string) {
    localStorage.removeItem(`${this.PREFIX}${linkId}`);
  }
  
  static getAll(): Record<string, string> {
    const tokens: Record<string, string> = {};
    
    for (let i = 0; i < localStorage.length; i++) {
      const key = localStorage.key(i);
      if (key?.startsWith(this.PREFIX)) {
        const linkId = key.substring(this.PREFIX.length);
        const token = localStorage.getItem(key);
        if (token) {
          tokens[linkId] = token;
        }
      }
    }
    
    return tokens;
  }
}
```

### Rate Limiting Handling

```typescript
class RateLimitedClient {
  private lastRequest = 0;
  private minInterval = 1000; // 1 second between requests
  
  async makeRequest(url: string, options: RequestInit = {}) {
    // Implement client-side rate limiting
    const now = Date.now();
    const timeSinceLastRequest = now - this.lastRequest;
    
    if (timeSinceLastRequest < this.minInterval) {
      await new Promise(resolve => 
        setTimeout(resolve, this.minInterval - timeSinceLastRequest)
      );
    }
    
    this.lastRequest = Date.now();
    
    const response = await fetch(url, options);
    
    // Handle server-side rate limiting
    if (response.status === 429) {
      const retryAfter = response.headers.get('Retry-After');
      const delay = retryAfter ? parseInt(retryAfter) * 1000 : 60000;
      
      throw new ApiError(
        `Rate limited. Retry after ${delay}ms`,
        'RATE_LIMITED',
        429
      );
    }
    
    return response;
  }
}
```

## Performance Optimization

### Caching Strategies

```typescript
class CachedLinkManager extends LinkManager {
  private cache = new Map<string, any>();
  private cacheExpiry = new Map<string, number>();
  private readonly CACHE_DURATION = 5 * 60 * 1000; // 5 minutes

  async getAnalytics(linkId: string, token: string, period = '7d') {
    const cacheKey = `analytics_${linkId}_${period}`;
    const now = Date.now();
    
    // Check cache
    if (this.cache.has(cacheKey)) {
      const expiry = this.cacheExpiry.get(cacheKey) || 0;
      if (now < expiry) {
        return this.cache.get(cacheKey);
      }
    }
    
    // Fetch fresh data
    const result = await super.getAnalytics(linkId, token, period);
    
    // Cache the result
    this.cache.set(cacheKey, result);
    this.cacheExpiry.set(cacheKey, now + this.CACHE_DURATION);
    
    return result;
  }
  
  clearCache() {
    this.cache.clear();
    this.cacheExpiry.clear();
  }
}
```
